00151 // using formulae from
00152 //  http://en.wikipedia.org/wiki/Cubic_interpolation#Interpolation_on_the_unit_interval_without_exact_derivatives
00153 // computes cubic interpolation of 4 points at point x,y
00154 // note: variable indicies off by +1, so centered at data1 == f(x=0)
00155 template <class DATA>
00156 inline PROMOTION(Real,DATA) CubicInterpolate (
00157         const Real& x,
00158         const DATA& data0, const DATA& data1, const DATA& data2, const DATA& data3
00159 )  __attribute__((always_inline));

00161 template <class DATA>
00162 inline PROMOTION(Real,DATA) CubicInterpolate (
00163         const Real& x,
00164         const DATA& data0, const DATA& data1, const DATA& data2, const DATA& data3
00165 )
00166 {
00167         const PROMOTION(Real,DATA) x2 = x * x;
00168 
00169         return Real(.5) * (
00170                 + (x*((2-x)*x-1))     * data0
00171                 + (x2*(3*x-5)+2)      * data1
00172                 + (x*((4-3*x)*x+1))   * data2
00173                 + ((x-1)*x2)          * data3);
00174 
00175 }
