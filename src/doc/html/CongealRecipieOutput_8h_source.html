00028 {
00029         typedef typename SOURCE::type_data DATA;
00030         static Real rSigma=ConfigValue("congeal.error[parzen].sigma",10.);
00031         static Real rApriori=ConfigValue("congeal.error[parzen].apriori",.2);
00032 
00033         // get ready to access
00034         for (int n=0; n<cSources; n++){
00035                 vsrc[n].PrepareForAccess();
00036         }
00037 
00038         Real rError=0;
00039                 
00040         for (int nSourceTest=0; nSourceTest<cSources; nSourceTest++){
00041                 Real rLogProbTotal=1.;
00042                 
00043                 // loop over all the points and measure the error
00044                 forpoint(TypeOfPoint(SOURCE),pt,0,vsrc[nSourceTest].Size()){
00045                         DATA dataTest;
00046 			::GetPoint(vsrc[nSourceTest],dataTest,pt);
00047 
00048                         Real rLogProbSample=0;
00049                         for (int nSource=0; nSource<cSources; nSource++){
00050                                 if (nSource==nSourceTest){
00051                                         continue;
00052                                 }
00053                                 DATA dataPrior;
00054 				::GetPoint(vsrc[nSource],dataPrior, pt);
00055 
00056                                 Real rProbSource = Gaussian(dataTest, dataPrior, rSigma);
00057 
00058                                 CONGEAL_ASSERTf(rProbSource>=0, "NEGATIVE ERROR %g", rProbSource);
00059 
00060                                 rLogProbSample += -log(rProbSource + rApriori);
00061                         }
00062 
00063                         rLogProbTotal +=rLogProbSample;
00064                 }
00065                 rLogProbTotal /= vsrc[nSourceTest].CSize();
00066                 rLogProbTotal /= cSources;

00068         }
00069
